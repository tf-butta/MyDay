// C++ code
int ledv = 8;    
int leda = 9;    
int ledr = 10;   


int btn1 = 2;    
int btn2 = 3;    
int btn3 = 4;    


int eAnterior1 = 0;
int eAnterior2 = 0;
int eAnterior3 = 0;

int ldrPin = A0;
int ledPin = 13;
int valorLuz;


int tempPin = A1;    
float temperatura;   

void setup()
{
 
  pinMode(btn1, INPUT);
  pinMode(btn2, INPUT);
  pinMode(btn3, INPUT);

  pinMode(ledv, OUTPUT);
  pinMode(leda, OUTPUT);
  pinMode(ledr, OUTPUT);

  pinMode(ledPin, OUTPUT);

  
  Serial.begin(9600);
}

void loop()
{
  
  int eActual1 = digitalRead(btn1);
  int eActual2 = digitalRead(btn2);
  int eActual3 = digitalRead(btn3);

  
  if (eActual1 == HIGH && eAnterior1 == LOW) {
    digitalWrite(ledv, HIGH);
    digitalWrite(leda, LOW);
    digitalWrite(ledr, LOW);
    delay(250);
  }
  eAnterior1 = eActual1;

  
  if (eActual2 == HIGH && eAnterior2 == LOW) {
    digitalWrite(ledv, HIGH);  
    digitalWrite(leda, HIGH);   
    digitalWrite(ledr, LOW);    
    delay(250);
  }
  eAnterior2 = eActual2;

  
  if (eActual3 == HIGH && eAnterior3 == LOW) {
    digitalWrite(ledv, LOW);
    digitalWrite(leda, LOW);
    digitalWrite(ledr, HIGH);
    delay(250);
  }
  eAnterior3 = eActual3;

  
  valorLuz = analogRead(ldrPin); 
  Serial.print("Valor de Luz: ");
  Serial.print(valorLuz);

  if (valorLuz < 500) {
    digitalWrite(ledPin, HIGH);
    Serial.println("  → Es de NOCHE ");
  } else {
    digitalWrite(ledPin, LOW);
    Serial.println("  → Es de DÍA ");
  }

  
  int lectura = analogRead(tempPin);
  temperatura = (lectura * 5.0 * 100.0) / 1024.0;  

  Serial.print("Temperatura: ");
  Serial.print(temperatura);
  Serial.println(" °C");

    
}
